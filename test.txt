#include <iostream>
#include <string>
#include <fstream>
#include <sstream>
#include <thread>

using namespace std;

int *board[9];
int row, col;
int sumH[9], sumV[9];
void is_row_ok();

int main()
{
	for (int i = 0; i < 9; ++i)
	{
		board[i] = new int[9];
	}

	string line;
	ifstream myFile("Testfile1.txt");

	for (int row = 0; row < 9; ++row)
	{
		string line;
		getline(myFile, line);
		if (!myFile.good())
			break;

		stringstream iss(line);
		cout << endl;

		for (int col = 0; col < 9; ++col)
		{
			string val;
			getline(iss, val, ',');
			if (!iss.good())
				break;

			stringstream convertor(val);
			convertor >> board[row][col];
			cout << board[row][col] << "  ";
		}
	}
	is_row_ok();

	cout << endl;
	return 0;
}

void is_row_ok()
{
	for (int i = 0; i < 9; ++i)
	{
		sumH[i] = 0;
		sumV[i] = 0;
		for (int j = 0; j < 9; ++j)
		{
			sumH[i] += board[i][j];
			sumV[i] += board[j][i];
			cout << endl;
			cout << "BOARD: " << board[i][j] << endl;
			cout << "I: " << i << endl;
			cout << "J: " << j << endl;
			cout << "sumH[i]:" << sumH[i] << endl;
		}
	}
}
