void is_row_ok()
{
	int row = 0;
	bool duplicate = false;

	while (row < 9)
	{
		int temp[9] = {};
		int refTemp[9] = {};

		for (int i = 0; i < row; ++i)
		{
			temp[9] = board[row][i];
			refTemp[9] = board[row][i];
		}
		sort(temp, temp + 9);

		for (int i = 0; i < 9; ++i)
		{
			for (int j = i + 1; j < 9; ++j)
			{
				if (temp[i] == refTemp[i])
				{
					duplicate = true;
				}
			}
		}
		break;
	}
}
